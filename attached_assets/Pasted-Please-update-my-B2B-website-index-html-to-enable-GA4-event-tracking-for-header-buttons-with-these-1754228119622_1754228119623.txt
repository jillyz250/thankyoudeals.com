**Important:** Only assign these IDs to the **header** links, not to any duplicate links in the page body.

---

2. **Add Header Button Click Tracking Script**

Right after the existing GA4 `<script>` tag in the `<head>` or just before `</body>`, add this:

<script>
  // Function to send GA4 events for header clicks
  function trackHeaderClick(eventName) {
    gtag('event', eventName, { location: 'Header' });
  }

  document.addEventListener("DOMContentLoaded", function() {
    // Map of header IDs to GA4 event names
    const headerLinks = {
      'Header_Retailers_Click': '#header-retailers',
      'Header_Advertisers_Click': '#header-advertisers',
      'Header_HowItWorks_Click': '#header-how',
      'Header_Demo_Click': '#header-demo',
      'Header_About_Click': '#header-about',
      'Header_ContactUs_Click': '#header-contact'
    };

    // Attach event listeners
    for (const [eventName, selector] of Object.entries(headerLinks)) {
      const el = document.querySelector(selector);
      if (el) {
        el.addEventListener('click', function() {
          trackHeaderClick(eventName);
        });
      }
    }
  });
</script>
